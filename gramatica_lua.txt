
Block => Estrela0

Estrela0 => Stmt; Estrela0
    |   VAZIO

Stmt => Vars = Exps
    |   Function
    |   do Block end
    |   while Exp do BLock end
    |   if Exp then Block Estrela1 Stmt' end  
    |   return Stmt3
    |   break
    |   for Stmt1
    |   local Stmt2
    |   function Name FunctionBody

Estrela1 => elseif Exp then Block Estrela1
    | VAZIO

Stmt' => (else Block)
    | VAZIO

Stmt1 => Name = Exp , Exp Stmt1' do Block end
    |   Names in Exps do Block end

Stmt1' => (, Exp)
    | VAZIO

Stmt2 => function Name FunctionBody
    |   Names = Exps

Stmt3 => Exps
    |   VAZIO

Exps => Exp Estrela2

Estrela2 => ,Exp Estrela2
    |   VAZIO

Exp => not Exp Exp'
    |   - Exp Exp'
    |   PrefixExp Exp'
    |   Function Exp'
    |   {T} Exp'
    |   nil Exp'
    |   true Exp'
    |   false Exp'
    |   Number Exp'
    |   String Exp'

 T => (Field Estrela3) 
    | VAZIO

Estrela3 => , Field Estrela3
    |   VAZIO

Exp' =>  BinOp Exp Exp' 
    |   VAZIO

PrefixExp => Name
    |   (Exp)
    |   Name PrefixExp'
    |   (Exp) PrefixExp'

PrefixExp' =>  [Exp]
    | [Exp] PrefixExp'

Field => [Epx] = Exp               
    |   Name = Exp

BinOp => or 
    | and 
    | < BinOp'
    | > BinOp'
    | ~= 
    | == 
    | . . 
    | + 
    | - 
    | * 
    | / 
    | ^ 

BinOp' => = | VAZIO    

Vars => Var Estrela4

Estrela4 => ,Var Estrela4
    |   VAZIO

Var => Name
    |   PrefixExp [Exp]

Function => FunctionBody
    |   FunctionBody Function

FunctionBody => Name (Params') Block end

Params' => Params | VAZIO

Params => Names

Names => Name Estrela5

Estrela5 => , Name Estrela5
    |   VAZIO
